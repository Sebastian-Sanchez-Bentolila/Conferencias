# Longitud de referencia (por ejemplo, la cantidad mxima de elementos en las listas)
max_len = max(len(col) for col in data.values())

# Asegurar que cada lista tenga la misma longitud
for key in data:
    if len(data[key]) < max_len:
        # Rellenar con cadenas vacas para igualar longitudes
        data[key].extend([""] * (max_len - len(data[key])))

# Filtrar anotaciones que mencionan escuchar msica
con_musica = df[df["Anotacion_1"].str.contains("msica", case=False)]
sin_musica = df[df["Anotacion_1"].str.contains("msica", case=False)]

# Conteo de ciclistas con y sin msica por motivo
con_musica_count = con_musica.groupby("Motivo")["Numero_de_ciclistas"].sum()
sin_musica_count = sin_musica.groupby("Motivo")["Numero_de_ciclistas"].sum()

# Crear grfico de barras apiladas
plt.figure(figsize=(10, 5))
plt.bar(con_musica_count.index, con_musica_count.values, label="Escuchando msica", color="lightgreen")
plt.bar(sin_musica_count.index, sin_musica_count.values, label="Sin msica", color="lightcoral", bottom=con_musica_count.values)
plt.title("Uso de Dispositivos de Entretenimiento por Motivo")
plt.xlabel("Motivo")
plt.ylabel("Cantidad de Ciclistas")
plt.legend()
